	\chapter{Tổng quan về đề tài}
	\section{Mô tả bài toán}
	
	Hackathon là một cuộc thi phát triển sản phẩm công nghệ trong một khoảng thời gian giới hạn, nơi các cá nhân hoặc nhóm cùng nhau xây dựng các giải pháp sáng tạo để giải quyết các vấn đề thực tiễn. Trong quá trình tổ chức Hackathon, ban tổ chức phải quản lý số lượng lớn thí sinh, đội thi, cố vấn, giám khảo và các bài dự thi.
	
	Việc quản lý thông tin bằng các phương pháp thủ công hoặc sử dụng nhiều nền tảng riêng biệt thường gây khó khăn trong việc theo dõi tiến độ, đánh giá kết quả và trao đổi thông tin giữa các bên liên quan. Điều này làm tăng khối lượng công việc cho ban tổ chức và dễ phát sinh sai sót trong quá trình vận hành cuộc thi.
	
	Để giải quyết những vấn đề trên, hệ thống SEAL Hackathon Management System được đề xuất nhằm hỗ trợ quản lý tập trung toàn bộ quá trình tổ chức và vận hành cuộc thi. Hệ thống cho phép người dùng đăng ký tham gia, tạo đội, quản lý dự án, nộp bài dự thi, nhận hỗ trợ từ cố vấn và tham gia quá trình đánh giá một cách hiệu quả.
	\section{Mục tiêu hệ thống}
	
	Hệ thống SEAL Hackathon Management System được xây dựng nhằm đạt được các mục tiêu sau:
	
	\begin{itemize}
		\item Quản lý tập trung thông tin cuộc thi Hackathon.
		\item Hỗ trợ đăng ký và quản lý người tham gia.
		\item Hỗ trợ tạo đội thi và quản lý thành viên.
		\item Quản lý quá trình nộp bài dự thi.
		\item Hỗ trợ kết nối giữa đội thi và cố vấn.
		\item Hỗ trợ giám khảo đánh giá và chấm điểm dự án.
		\item Cung cấp các chức năng quản trị và thống kê phục vụ ban tổ chức.
	\end{itemize}
	\section{Phạm vi hệ thống}
	
Hệ thống tập trung hỗ trợ các hoạt động chính trong cuộc thi Hackathon từ giai đoạn đăng ký tham gia đến khi công bố kết quả.

Các chức năng được hỗ trợ bao gồm:

\begin{itemize}
	\item Đăng ký và đăng nhập tài khoản.
	\item Quản lý hồ sơ cá nhân.
	\item Đăng ký tham gia cuộc thi.
	\item Tạo và quản lý đội thi.
	\item Quản lý cố vấn.
	\item Nộp và cập nhật bài dự thi.
	\item Đánh giá và chấm điểm dự án.
	\item Quản trị cuộc thi và thống kê báo cáo.
\end{itemize}

Các chức năng ngoài phạm vi của hệ thống bao gồm quản lý thanh toán, livestream sự kiện và các hoạt động tài chính của cuộc thi.
	
	\chapter{Phân tích yêu cầu}
	\section{Các đối tượng sử dụng}
	
	Hệ thống SEAL Hackathon Management System bao gồm các đối tượng sử dụng chính sau:
	
	\begin{itemize}
		\item \textbf{Participant}: Người tham gia cuộc thi, có thể đăng ký tài khoản, đăng ký tham gia Hackathon và theo dõi các hoạt động liên quan đến cuộc thi.
		
		\item \textbf{Team Member}: Thành viên của đội thi, tham gia xây dựng dự án, quản lý nhóm và thực hiện nộp bài dự thi.
		
		\item \textbf{Mentor}: Chuyên gia hoặc cố vấn hỗ trợ các đội thi trong quá trình phát triển dự án và giải quyết các vấn đề chuyên môn.
		
		\item \textbf{Judge}: Giám khảo chịu trách nhiệm đánh giá, chấm điểm và đưa ra nhận xét đối với các bài dự thi.
		
		\item \textbf{Admin}: Quản trị viên chịu trách nhiệm quản lý người dùng, cuộc thi, đội thi, bài nộp và toàn bộ hoạt động của hệ thống.
	\end{itemize}

	\section{Phác thảo giao diện}
	Mục này trình bày phác thảo thiết kế giao diện (Mockup) cho các chức năng cốt lõi của hệ thống SEAL Hackathon Management System. Giao diện được thiết kế theo hướng hiện đại, trực quan, tập trung vào trải nghiệm người dùng (UX/UI) và đảm bảo tính khả dụng trên nhiều thiết bị.
	
	\subsection{Giao diện Chi tiết Cuộc thi Hackathon}
	Giao diện Chi tiết Cuộc thi (Hackathon Details) là trang thông tin trung tâm, cung cấp cái nhìn toàn diện cho \textit{Participant} về thể lệ, tiến trình và quyền lợi khi tham gia SEAL Hackathon.
	\begin{itemize}
		\item \textbf{Khối thông tin tổng quan (Hero Section):} Hiển thị hình ảnh thương hiệu, trạng thái cuộc thi cùng nút hành động "Đăng ký tham gia ngay" (Call-to-Action).
		\item \textbf{Khối Lộ trình thời gian (Timeline Widget):} Hiển thị tiến trình cuộc thi dưới dạng trục thời gian với các mốc quan trọng (Mở cổng đăng ký, Khai mạc, Hạn chót nộp bài, Chung kết).
		\item \textbf{Khối Thể lệ và Tiêu chí:} Trình bày chi tiết quy định cuộc thi (yêu cầu kho lưu trữ mã nguồn mở) và tiêu chí đánh giá của Giám khảo.
	\end{itemize}
	
	\subsection{Giao diện Quản lý Đội nhóm (Team Management)}
	Phân hệ này là không gian làm việc chuyên biệt để các thí sinh thiết lập bộ máy nhân sự nội bộ cho cuộc thi, bao gồm:
	\begin{itemize}
		\item \textbf{Khối Thiết lập Đội thi:} Cung cấp hai biểu mẫu xử lý song song là Tạo đội mới (trả về mã mời) và Tham gia đội (nhập mã mời có sẵn). Hệ thống giao tiếp với các API \texttt{/teams/create} và \texttt{/teams/join}.
		\item \textbf{Bảng Danh sách Nhân sự:} Phân tách làm hai khu vực gồm "Thành viên hiện tại" và "Thành viên chờ duyệt". Trưởng nhóm có quyền thao tác trực tiếp (Đồng ý/Từ chối) trên các yêu cầu gia nhập thông qua API \texttt{/teams/approve}.
	\end{itemize}
	
	\subsection{Giao diện Dashboard và Nộp bài (Dashboard \& Submission)}
	Màn hình Dashboard là trung tâm tương tác chính, được xây dựng bằng cấu trúc lưới và thẻ (Grid \& Card layout) nhằm tối ưu hóa không gian hiển thị.
	\begin{itemize}
		\item \textbf{Bố cục tổng quan:} Gồm Sidebar điều hướng bên trái và thanh tiêu đề cấu hình Token xác thực phía trên.
		\item \textbf{Khối Nộp bài thi (Project Submission):} Thiết kế biểu mẫu đóng gói sản phẩm thông qua trường nhập liệu URL kho lưu trữ GitHub (\texttt{github-link}). Thí sinh có thể liên tục cập nhật đường dẫn trước thời điểm đóng cổng nộp bài thông qua nút "Nộp bài / Cập nhật" (API \texttt{/submissions/submit}).
		\item \textbf{Cơ chế phản hồi (Toast UI):} Mọi tương tác bất đồng bộ đều được phản hồi trực quan bằng các thông báo ngắn (Toast Message) báo lỗi hoặc thành công, giúp thí sinh kiểm soát tốt quá trình thao tác.
	\end{itemize}

# Các thư viện chính:
# - react: ^18.2.0

# - lucide-react: ^0.263.1 (icon library)
# - vite: ^4.3.9 (build tool)
# - tailwindcss: ^3.3.2 (CSS framework)

#### Chạy Backend
cd backend
python -m uvicorn app.main:app --reload
# Server chạy tại http://localhost:8000
# API Docs tại http://localhost:8000/docs

#### Chạy Frontend
npm run dev
# Frontend chạy tại http://localhost:5173 (mặc định của Vite)

#### Public server qua Ngrok (tùy chọn)
ngrok http 8000

### 3.2. Các thành phần chính

#### 3.2.1. FastAPI Framework
- **Mục đích**: Xây dựng REST API nhanh chóng với automatic validation
- **Tính năng**:
  - Automatic API documentation (Swagger UI tại `/docs`)
  - Data validation với Pydantic
  - Type hints hỗ trợ
  - Async/await support

#### 3.2.2. Uvicorn Server
- **Mục đích**: ASGI server để chạy ứng dụng FastAPI
- **Chức năng**: Xử lý HTTP requests, hỗ trợ async operations

#### 3.2.3. SQLAlchemy ORM
- **Mục đích**: Object-Relational Mapping để thao tác database
- **Models** (trong `models.py`):
  - **User**: Thông tin người dùng (username, password_hash, role)
  - **Team**: Thông tin đội thi (team_name, leader_id, members)
  - **Submission**: Bài nộp dự án (team_id, project_name, description, score)

#### 3.2.4. Authentication & Authorization
- **JWT (JSON Web Token)**: Xác thực người dùng
  - Tạo token khi đăng nhập
  - Validate token cho các protected routes
- **Password Hashing**: Sử dụng bcrypt để mã hóa mật khẩu
- **Roles**: Phân quyền (Candidate, Jury, Admin)

#### 3.2.5. Pydantic
- **Mục đích**: Validation dữ liệu đầu vào
- **Schemas**: Định nghĩa cấu trúc dữ liệu cho request/response
  - UserCreate, UserLogin
  - TeamCreate, TeamResponse
  - SubmissionCreate, SubmissionResponse

### 3.3. API Endpoints

#### Authentication (`/auth`)
- `POST /auth/register`: Đăng ký tài khoản mới
- `POST /auth/login`: Đăng nhập và nhận JWT token

#### Teams (`/teams`)
- `POST /teams/create`: Tạo đội mới
- `POST /teams/join`: Tham gia đội
- `POST /teams/approve`: Phê duyệt thành viên đội

#### Submissions (`/submissions`)
- `POST /submissions/submit`: Nộp bài dự án
- `GET /submissions/all`: Lấy danh sách bài nộp (cho Jury)
- `POST /submissions/score`: Chấm điểm bài nộp (cho Jury)

	\section{Thiết kế cơ sở dữ liệu}
	Để đảm bảo hệ thống SEAL Hackathon Management System vận hành ổn định, lưu trữ dữ liệu nhất quán và tối ưu hóa tốc độ truy vấn, cơ sở dữ liệu được thiết kế theo mô hình dữ liệu quan hệ dưới dạng chuẩn hóa (3NF). Hệ thống quản lý tập trung các luồng thông tin từ tài khoản người dùng, hồ sơ đội thi, các mốc thời gian nộp bài cho đến tiêu chí chấm điểm của giám khảo.

	\section{Mô tả các thực thể và mối quan hệ}
	Dựa trên sơ đồ ERD, cấu trúc dữ liệu của hệ thống được phân tách thành các thực thể và mối quan hệ cụ thể như sau:
	
	\subsection{Mô tả các thực thể chi tiết}
	\begin{itemize}
		\item \textbf{Users (Người dùng):} Lưu trữ thông tin định danh toàn bộ người dùng trong hệ thống.
		\begin{itemize}
			\item \textit{Thuộc tính:} \texttt{UserID} (Khóa chính), \texttt{FullName}, \texttt{Email}, \texttt{PasswordHash}, \texttt{Role} (Participant, Mentor, Judge, Admin), \texttt{CreatedAt}.
		\end{itemize}
		
		\item \textbf{Hackathons (Cuộc thi):} Lưu trữ thông tin về các cuộc thi Hackathon do Admin khởi tạo.
		\begin{itemize}
			\item \textit{Thuộc tính:} \texttt{HackathonID} (Khóa chính), \texttt{Title}, \texttt{Description}, \texttt{StartDate}, \texttt{EndDate}, \texttt{Status}.
		\end{itemize}
		
		\item \textbf{Teams (Đội thi):} Quản lý thông tin của các nhóm tham gia cuộc thi.
		\begin{itemize}
			\item \textit{Thuộc tính:} \texttt{TeamID} (Khóa chính), \texttt{TeamName}, \texttt{InviteCode}, \texttt{LeaderID} (Khóa ngoại liên kết tới Users), \texttt{HackathonID} (Khóa ngoại liên kết tới Hackathons).
		\end{itemize}
		
		\item \textbf{Submissions (Bài dự thi):} Lưu trữ thông tin các sản phẩm/mã nguồn do các đội thi nộp lên hệ thống.
		\begin{itemize}
			\item \textit{Thuộc tính:} \texttt{SubmissionID} (Khóa chính), \texttt{TeamID} (Khóa ngoại), \texttt{GithubURL}, \texttt{ProjectDescription}, \texttt{SubmittedAt}.
		\end{itemize}
		
		\item \textbf{Evaluations (Đánh giá/Chấm điểm):} Lưu trữ điểm số và nhận xét của giám khảo dành cho các bài dự thi.
		\begin{itemize}
			\item \textit{Thuộc tính:} \texttt{EvaluationID} (Khóa chính), \texttt{SubmissionID} (Khóa ngoại), \texttt{JudgeID} (Khóa ngoại liên kết tới Users), \texttt{Score}, \texttt{Comment}, \texttt{EvaluatedAt}.
		\end{itemize}
		
		\item \textbf{Mentorships (Cố vấn):} Bản ghi trung gian thể hiện sự kết nối, hỗ trợ giữa Mentor và các Đội thi.
		\begin{itemize}
			\item \textit{Thuộc tính:} \texttt{MentorshipID} (Khóa chính), \texttt{MentorID} (Khóa ngoại liên kết tới Users), \texttt{TeamID} (Khóa ngoại), \texttt{AssignedAt}.
		\end{itemize}
	\end{itemize}
	
	\subsection{Mô tả các mối quan hệ}
	\begin{itemize}
		\item \textbf{Users -- Teams (1 - N):} Một người dùng làm trưởng nhóm (\texttt{LeaderID}) chỉ quản lý 1 đội thi tại một thời điểm, nhưng một tài khoản người dùng nói chung có thể đăng ký tham gia vào nhiều đội thi với vai trò thành viên thông qua bảng liên kết thành viên.
		\item \textbf{Hackathons -- Teams (1 - N):} Một cuộc thi Hackathon có thể có nhiều đội thi đăng ký tham gia, nhưng một đội thi chỉ thuộc về một cuộc thi duy nhất.
		\item \textbf{Teams -- Submissions (1 - 1):} Để đảm bảo tính công bằng, mỗi đội thi chỉ được phép có duy nhất một bài dự thi chính thức trên hệ thống (hoặc cập nhật đè trên bài cũ).
		\item \textbf{Submissions -- Evaluations (1 - N):} Một bài dự thi của một đội có thể được chấm điểm và nhận xét bởi nhiều giám khảo khác nhau (\texttt{JudgeID}) để lấy điểm trung bình làm kết quả cuối cùng.
		\item \textbf{Users (Mentor) -- Teams (N - N):} Một cố vấn có thể hỗ trợ nhiều đội thi khác nhau và ngược lại, một đội thi có thể nhận sự cố vấn từ nhiều chuyên gia thông qua bảng trung gian \texttt{Mentorships}.
	\end{itemize}
